COPING THE EXISTING TABLE TO ANOTHER TABLE
==============================================

CREATE TABLE EMP2 AS EMP WITH DATA

CREATE TABLE EMP2 AS SELECT * FROM EMP;

SELECT * FROM EMP2;--ALL ROWS RETRIVED

CREATE TABLE EMP3 AS EMP WITH STRUCTURE ONLY (DONT COPY DATA)

CREATE TABLE EMP3 AS SELECT * FROM EMP WHERE 1=2;


SELECT * FROM EMP3;--NO ROWS RETRIVED

CREATE TABLE EMP4 WITH THE COLUMNS OF EMPNO,ENAME,SAL,DEPTNO

CREATE TABLE EMP4 AS SELECT EMPNO,ENAME,SAL,DEPTNO FROM EMP ;

SELECT * FROM EMP4
