Using Rownum:
=========
TOP 1ST SALARY:
---------------

ORDER THE SALARY BY DESCENDING ORDER THE PICK THE 1ST ROW

SELECT * FROM EMP ORDER BY SAL DESC WHERE ROWNUM=1;

ERROR: ORDER BY CLAUSE SHOULD BE THE LAST CLAUSE

--SELECT EMPNO,ENAME FROM (SELECT EMPNO,ENAME,SAL,DEPTNO FROM EMP)


SELECT  * FROM (SELECT EMPNO,ENAME,SAL FROM EMP ORDER BY SAL DESC) WHERE ROWNUM=1

TOP 1ST ,2ND,3RD

SELECT  * FROM (SELECT EMPNO,ENAME,SAL FROM EMP ORDER BY SAL DESC) WHERE ROWNUM<=3
